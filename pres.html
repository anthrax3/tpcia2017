kritika.forEachModifiedFile(function(file) {
    if (/[A-Z]/.test(file.basename)) {
        kritika.addViolation(file, {
            message: 'Uppercase in names is forbidden!'
        });
    }

    if (/# TODO/.test(file.slurp())) {
        kritika.addViolation(file, {
            message: 'TODOs are forbidden!'
        });
    }
});
